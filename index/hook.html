Methods and Properties

editor.fire(event, data)

Fire a hook.

editor.fire('click', [datum]);

editor.hooks

Return the added hooks.

console.log(editor.hooks);

editor.off(event, then)

Remove a hook.

editor.off('click');
editor.off('click', onClick); // With context

editor.on(event, then)

Add a new hook.

editor.on('click', () => {
    alert('Click!');
});
function onClick() {
    alert('Click!');
}

editor.on('click', onClick); // With context
