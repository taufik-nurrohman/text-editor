let editor = new TE(self, state = {
        tab: '\t',
        source: {
            close: {
                '(': ')',
                '{': '}',
                '[': ']',
                '"': '"',
                "'": "'",
                '<': '>'
            },
            // Custom function to trigger the outdent
            pull: function(e) {
                let isCtrl = e.ctrlKey,
                    key = e.key,
                    keyCode = e.keyCode;
                return isCtrl && ((key && '[' === key) || (keyCode && 219 === keyCode));
            },
            // Custom function to trigger the indent
            push: function(e) { … }
            select: true
        }
    });
  • state.tab → The default indent character.
  • state.source.close → List of token on the left that should be “auto-closed” by token on the right.
  • state.source.select → Set to false to disable the smart word selection feature.
