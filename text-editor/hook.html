Methods and Properties

editor.fire(name, lot)

Trigger hooks.

editor.fire('click', [data]);

editor.hooks

Return the added hooks.

console.log(editor.hooks);

editor.off(name, fn)

Remove a hook.

editor.off('click');
editor.off('click', onClick); // With context

editor.on(name, fn)

Add a new hook.

editor.on('click', data => {
    alert('Click!');
});
function onClick(data) {
    alert('Click!');
}

editor.on('click', onClick); // With context
